import math

# Вводим координаты точки M
x = float(input("Введите координату x точки M: "))
y = float(input("Введите координату y точки M: "))

# Вводим координаты центра круга и его радиус
x0 = float(input("Введите координату x0 центра круга: "))
y0 = float(input("Введите координату y0 центра круга: "))
radius = float(input("Введите радиус круга: "))

# Вычисляем расстояние от точки M до центра круга
distance = math.sqrt((x - x0)**2 + (y - y0)**2)

# Проверяем, попадает ли точка в круг
if distance <= radius:
    print("Точка M попадает в круг.")
else:
    print("Точка M не попадает в круг.")

import math

# Вводим натуральное число n
n = int(input("Введите натуральное число n: "))

# Инициализируем переменную для суммы
sum_result = 0
current_sin_sum = 0

# Вычисляем сумму 1/sin(1) + 1/(sin(1+2)) + ... + 1/(sin(1+2+...+n))
for i in range(1, n + 1):
    current_sin_sum += math.sin(i)
    sum_result += 1 / current_sin_sum

# Выводим результат
print(' сумма равна: ',sum_result)

import math

# Вводим действительное число x и натуральное число n
x = float(input("Введите действительное число x: "))
n = int(input("Введите натуральное число n: "))

# Инициализируем результат
result = 0

# Вычисляем сумму x^i / i! для i от 1 до n
for i in range(1, n + 1):
    result += (x ** i) / math.factorial(i)

# Выводим результат
print('Результат выражения x^1/1! + x^2/2! + ... + x^n/n! равен:' ,result)
